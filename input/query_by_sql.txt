SELECT department_id, COUNT(product_id) AS number_of_orders, 
SUM(reordered) AS number_of_first_orders, 
(SUM(reordered)\COUNT(product_id)) AS percentage
(SELECT p.department_id AS department_id, op.product_id AS product_id, op.reordered AS reordered 
FROM order_products LEFT JOIN products AS p ON op.product_id = p.product_id)
GROUP BY product_id, department_id
ORDER BY department_id
